代码

clc
m=-1.5;
Nx=50;
Ny=200;
kxarray=2*pi/Nx*((-Nx/2-1/2):1:(Nx/2-1/2));
kyarray=2*pi/Ny*((-Ny/2-1/2):1:(Ny/2-1/2));
MB = zeros(2*Nx);
for iky=1:length(kyarray)
    ky=kyarray(iky);
    A1 = m+cos(ky);
    A2 = 1i*sin(ky);
    A3 = -1i*sin(ky);
    A4 = -(m+cos(ky));
    A = [A1,A2;A3,A4];
    for j = 1:Nx
        MA((2*j-1):(2*j),(2*j-1):(2*j)) = A;
    end
    
    B1 = 1/2;
    B2 = 1/(2*1i);
    B3 = 1/(2*1i);
    B4 = -1/2;
    B = [B1,B2;B3,B4];
    
    for j = 1:Nx-1
        MB((2*j+1):(2*j+2),(2*j-1):(2*j)) = B;
    end
    
    H = MA+MB+MB';
   
    [V,D] = eig(H);
    F(:,iky) = diag(D);
end

figure
plot(kyarray,F,'k')

%一维橫场XY模型的缠绕数

clear
N=10000;
karray=2*pi/(N)*(-(N/2-1/2):1:(N/2-1/2));
ks=karray;
dk=karray(2)-karray(1);

harray=[0.0:0.1:1.0];
gammaarray=[0:0.1:1.0];

J=1;

% 方法一（数值求和）
tic
for ig=1:length(gammaarray)
    gamma=gammaarray(ig);
    for ih=1:length(harray)
        h=harray(ih); 
        wn(ih,ig)=0;
        for ik=1:length(karray)
            k=karray(ik);
            Bz=J*cos(k)+h;dBz=-J*sin(k);
            By=J*gamma*sin(k);dBy=J*gamma*cos(k);
            wn(ih,ig)=wn(ih,ig)+(dk*(Bz*dBy-By*dBz)/(By^2+Bz^2))/(2*pi);
        end
    end
end
disp(['方法一（数值求和）用时：' num2str(round(toc*10)/10) 's.'])
% 方法一（数值求和）
clear wn By Bz dBy dBz
tic
for ig=1:length(gammaarray)
    gamma=gammaarray(ig);
    for ih=1:length(harray)
        h=harray(ih);
        Bz=J*cos(ks)+h;dBz=-J*sin(ks);
        By=J*gamma*sin(ks);dBy=J*gamma*cos(ks);
        wn(ih,ig)=sum(dk*(Bz.*dBy-By.*dBz)./(By.^2+Bz.^2))/(2*pi);
    end
end
disp(['方法二（矩阵求和）用时：' num2str(round(toc*10)/10) 's.'])
% 方法三 
clear wn
tic
for ig=1:length(gammaarray)
    delta1=gammaarray(ig);
    for ih=1:length(harray)
        h=harray(ih);
        
        ising=@(x)(((J*cos(x)+h).*(J*gamma*cos(x))-(J*gamma*sin(x)).*(-J*sin(x)))./((J*gamma*sin(x)).^2+(J*cos(x)+h).^2))/(2*pi);
        wn(ih,ig)=integral(ising,-pi,pi);
    end
end
disp(['方法三（积分求和）用时：' num2str(round(toc*10)/10) 's.'])

态密度

参考：态密度

二维立方晶格模型的色散关系 \[ \varepsilon_{k}=-2 t\left[\cos \left(k_{x}+\cos \left(k_{y}\right)\right]\right. \] 态密度定义为单位频率间隔内的状态数 \[ g(\varepsilon)=\lim _{\Delta \varepsilon \rightarrow 0} \frac{\Delta n}{\Delta \varepsilon}=\frac{d n}{d \varepsilon} \]

Matlab

% DOS 态密度
% tight binding parameters
eps = 0;
t = 1;

% number of k values per dimension
lv = 5000;
% number of bins for the DOS
nbins = 101;

% vector of k values
k=linspace(-pi,pi,lv);
[kx,ky] = meshgrid(k,k);

% calculate the energy
E = eps - 2 * t * (cos(kx)+cos(ky));

% bin the energy to get the DOS,what it is in python???
[DOS, Eval] = hist(E(:),nbins);

% 归一化鸭！！！
DOS = DOS/(Eval(2)-Eval(1))/lv.^2;

% 这里把自旋也考虑进来，其实没多大影响。
%DOS = DOS * 2;

plot(Eval,DOS)

python

import matplotlib.pyplot as plt
import numpy as np
#parameters
mu=0
t=1
kx=0
ky=0
delta=(2*np.pi)/1000
res=[]
for kx in np.linspace(-np.pi,np.pi,1001):
    for ky in np.linspace(-np.pi,np.pi,1001):
        E=mu-2*t*(np.cos(kx)+np.cos(ky))
        res.append(E)
        ky=ky+delta
        kx=kx+delta

plt.title('DOS of 2d TB Model')
plt.xlabel('E')
plt.ylabel('DOS')
#density=Ture for normalization
plt.hist(res,bins=1000,density=1)
plt.show()

m=-1.5;
Nx=50;

    ky=kyarray(iky);
    A1 = m+cos(ky);
    A2 = 1i*sin(ky);
    A3 = -1i*sin(ky);
    A4 = -(m+cos(ky));
    A = [A1,A2;A3,A4];

        MA((2*j-1):(2*j),(2*j-1):(2*j)) = A;

    end

clear
N=10000;
karray=2*pi/(N)*(-(N/2-1/2):1:(N/2-1/2));
ks=karray;
dk=karray(2)-karray(1);

harray=[0.0:0.1:1.0];
gammaarray=[0:0.1:1.0];

J=1;

% 方法一（数值求和）
tic
for ig=1:length(gammaarray)
    gamma=gammaarray(ig);
    for ih=1:length(harray)
        h=harray(ih);

        for ik=1:length(karray)
            k=karray(ik);
            Bz=J*cos(k)+h;dBz=-J*sin(k);
            By=J*gamma*sin(k);dBy=J*gamma*cos(k);
            wn(ih,ig)=wn(ih,ig)+(dk*(Bz*dBy-By*dBz)/(By^2+Bz^2))/(2*pi);
        end
    end
end
disp(['方法一（数值求和）用时：' num2str(round(toc*10)/10) 's.'])
% 方法一（数值求和）

    end
end

态密度

参考：态密度

二维立方晶格模型的色散关系 \[ \varepsilon_{k}=-2 t\left[\cos \left(k_{x}+\cos \left(k_{y}\right)\right]\right. \] 态密度定义为单位频率间隔内的状态数 \[ g(\varepsilon)=\lim _{\Delta \varepsilon \rightarrow 0} \frac{\Delta n}{\Delta \varepsilon}=\frac{d n}{d \varepsilon} \]

Matlab

% DOS 态密度
% tight binding parameters
eps = 0;
t = 1;

% number of k values per dimension
lv = 5000;
% number of bins for the DOS
nbins = 101;

% vector of k values
k=linspace(-pi,pi,lv);
[kx,ky] = meshgrid(k,k);

% calculate the energy
E = eps - 2 * t * (cos(kx)+cos(ky));

% bin the energy to get the DOS,what it is in python???
[DOS, Eval] = hist(E(:),nbins);

% 归一化鸭！！！
DOS = DOS/(Eval(2)-Eval(1))/lv.^2;

Kane-Mele Model - mathematica

Kane-Mele Model能带的mathematica代码

布拉维格子的正格矢 Direct lattice \[ \vec{a}_{1}=\frac{a}{2}(\hat{x}+\sqrt{3} \hat{y}), \quad \vec{a}_{2}=\frac{a}{2}(-\hat{x}+\sqrt{3} \hat{y}) \]

对应的倒格矢 Reciprocal lattice: \[ \vec{b}_{1}=\frac{1}{a}\left(\hat{x}+\frac{\hat{y}}{\sqrt{3}}\right), \quad \vec{b}_{2}=\frac{1}{a}\left(-\hat{x}+\frac{\hat{y}}{\sqrt{3}}\right) \]

石墨烯中的紧束缚模型 \[ H=-t \sum_{\langle i \alpha, j \beta\rangle}\left(c_{i \alpha}^{\dagger} c_{j \beta}+\text { h.c. }\right)+\sum_{i \alpha} m_{\alpha} c_{i \alpha}^{\dagger} c_{i \alpha} \] 式中

  • \(i, j\) 代表原胞的位置
  • \(\alpha, \beta=1,2\) 代表了原胞中两个原子。
  • 这里令 \(m_{\alpha}=(-1)^{\alpha} m\) 加入 了原胞中两原子的差异性。

傅里叶变换 \[ c_{i \alpha}^{\dagger}=\frac{1}{\sqrt{N}} \sum_{ k \alpha} c_{ k \alpha}^{\dagger} e^{-i k \cdot R _{i}} \] 仅考虑最邻近作用 \[ \begin{aligned} H=&-t \sum_{i}\left(c_{i 1}^{\dagger} c_{i 2}+c_{i 1}^{\dagger} c_{i-\vec{a}_{1}, 2}+c_{i 1}^{\dagger} c_{i-\vec{a}_{2}, 2}+\text { h.c. }\right)+\sum_{i \alpha} m_{\alpha} c_{i \alpha}^{\dagger} c_{i \alpha} \\ =&-t \sum_{ k }\left[c_{ k 1}^{\dagger} c_{ k 2}\left(1+e^{-i k \cdot \vec{a}_{1}}+e^{-i k \cdot \vec{a}_{2}}\right)+c_{ k 2}^{\dagger} c_{ k 1}\left(1+e^{i k \cdot \vec{a}_{1}}+e^{i k \cdot \vec{a}_{2}}\right)\right] \\ &+\sum_{ k \alpha} m_{\alpha} c_{ k \alpha}^{\dagger} c_{ k \alpha} \end{aligned} \]

新表示

引入代表原胞中两个原子的二自由度雁自旋 \(\vec{\sigma}\) \[ \sigma^{+}=\left(\begin{array}{ll} 0 & 2 \\ 0 & 0 \end{array}\right), \sigma^{-}=\left(\begin{array}{ll} 0 & 0 \\ 2 & 0 \end{array}\right) \]

\[ \sigma^{\pm}=\sigma^{x} \pm i \sigma^{y} \]

哈密顿量可以写成 \[ \begin{aligned} H=&-\frac{t}{2} \sum_{ k \alpha \beta}\left[\sigma_{\alpha \beta}^{+} c_{ k \alpha}^{\dagger} c_{ k \beta}\left(1+e^{-i k \cdot \vec{a}_{1}}+e^{-i k \cdot \vec{a}_{2}}\right)+\sigma_{\alpha \beta}^{-} c_{ k \alpha}^{\dagger} c_{ k \beta}\left(1+e^{i k \cdot \vec{a}_{1}}+e^{i k \cdot \vec{a}_{2}}\right)\right] \\ &+m \sum_{ k \alpha} \sigma_{\alpha \alpha}^{z} c_{ k \alpha}^{\dagger} c_{ k \alpha} \end{aligned} \] 为了让计算更为简洁，并引入一种有用的技巧，定义 \[ \begin{aligned} d^{z}( k ) &=m \\ d^{\pm}( k ) &=-t\left(1+e^{\mp i k \cdot \vec{a}_{1}}+e^{\mp i k \cdot \vec{a}_{2}}\right) \\ d^{x}( k ) &=\frac{1}{2}\left(d^{+}( k )+d^{-}( k )\right)=-t\left(1+\cos k \cdot \vec{a}_{1}+\cos k \cdot \vec{a}_{2}\right) \\ d^{y}( k ) &=\frac{1}{2 i}\left(d^{+}( k )-d^{-}( k )\right)=-t\left(\sin k \cdot \vec{a}_{1}+\sin k \cdot \vec{a}_{2}\right) \end{aligned} \] 考虑到 \(\sigma^{\pm} d^{\pm}( k )=d^{x}( k ) \sigma^{x}+d^{y}( k ) \sigma^{y}\), 哈密顿量可以写成 \[ H=\sum_{ k , \alpha, \beta} d ( k ) \cdot \vec{\sigma}_{\alpha \beta} c_{ k \alpha}^{\dagger} c_{ k \beta} \] 要计算其本征值，只需要对角化 \[ H( k )= d ( k ) \cdot \vec{\sigma} \] 考虑到泡利自旋矩阵的特性 \(\sigma^{i} \sigma^{j}=\epsilon_{i j k} \sigma^{k},(i \neq j), \sigma^{i} \sigma^{i}= 1 ,(i=j)\) \[ H( k )^{2}=d^{i} d^{j} \sigma^{i} \sigma^{j}=d^{i} d^{i}+\epsilon_{i j k} d^{i} d^{j} \sigma^{k}= d ( k ) \cdot d ( k ) \] 其中第二项因其为对称张量和反对称张量的乘积为零。因 \(H( k )^{2}\) 对角，本征值可以容易地看出 \[ \epsilon_{\pm}( k )=\pm \sqrt{ d ( k ) \cdot d ( k )} \]

狄拉克锥 Kane-Mele Model

作为一个广为人知的结论，当原胞中的原子相同，即 \(m=0\) 时，石墨烯的能带中将出现狄拉克雉，狄 拉克雉的顶点被称作狄拉克点，可以由能带的零点给出。在本文的坐标下，狄拉克点的坐标为 : \[ \begin{aligned} k_{x} &=\frac{4 \pi}{3 a}, k_{y}=0 \\ k_{x} &=\frac{-4 \pi}{3 a}, k_{y}=0 \end{aligned} \] 在这两点附近展开 \((9)\) 可得到狄拉克点附近的哈密顿量 \[ H_{+}( k )=v\left(k_{x} \sigma^{x}+k_{y} \sigma^{y}\right)+m \sigma^{z} \] \[ H_{-}( k )=v\left(-k_{x} \sigma^{x}+k_{y} \sigma^{y}\right)+m \sigma^{z} \] 其中 \(v=\frac{\sqrt{3} t a}{2}\), 类似于上面引入 \(\vec{\sigma}\), 引入代表两个狄拉克点的谷自旋 \(\vec{\tau}\), 上式可以写成 \[ H( k )=v\left(k_{x} \tau^{z} \sigma^{x}+k_{y} \sigma^{y}\right)+m \sigma^{z} \] 这就是 Kane-Mele Model

对称性

  • 宇称变换 \(( P : k \rightarrow- k )\)
  • 时间反演 \(( T : k \rightarrow- k , \quad S \rightarrow- S )\)

宇称变换通俗来说是一个空间反演变换，在这个变换下原胞中的两原子将互换，所以 \(\vec{\sigma}\) 将由下式给 出: \[ P : \vec{\sigma} \rightarrow\left(\begin{array}{ll} 0 & 1 \\ 1 & 0 \end{array}\right) \vec{\sigma}\left(\begin{array}{ll} 0 & 1 \\ 1 & 0 \end{array}\right) \] 所以有 \(P : \sigma^{x} \rightarrow \sigma^{x}, \sigma^{y} \rightarrow-\sigma^{y}, \sigma^{z} \rightarrow-\sigma^{z}\) 动量空间的反演显然会让两个狄拉克点交换，故 \(\vec{\tau}\) 在 \(P\) 下变换同上。而时间反演下动量的变换关系和宇称变换相同，故 \(\vec{\tau}\) 的变换也相同。时间反演下原胞 中两原子显然不会变化，故 \(\vec{\sigma}\) 不变。电子自旋 \(\vec{S}\) 显然在宇称变换下不变。以上总结为表。 \[ \begin{array}{c|c|c} & P & T \\ \hline \sigma^{x} & \sigma^{x} & \sigma^{x} \\ \sigma^{y} & -\sigma^{y} & \sigma^{y} \\ \sigma^{z} & -\sigma^{z} & \sigma^{z} \\ \tau^{z} & -\tau^{z} & -\tau^{z} \\ S^{z} & S^{z} & -S^{z} \end{array} \] 猜到一个满足PT联合对称的自旋轨道耦合哈密顿量： \[ \sigma^{z} \tau^{z}\left(S^{z}\right)_{\alpha \beta} \]

次近邻&自旋轨道耦合

\[ H_{S O}=i t_{2} \sum_{\langle\langle i, j\rangle\rangle \alpha \beta} \nu_{i j}\left(S^{z}\right)_{\alpha \beta} c_{i \alpha}^{\dagger} c_{j \beta} \]

其中

  • \(v_{i j}=\pm 1\) 取决于跃迁时是向左抛还是向右抛 \((\) 图中即为左抛 \()\), 这对应于电子顺逆时针的旋转。
  • 由于A位原子得到次近邻仍为A位原子，且容易看出 \(A , B\) 两位置的对应的次近邻跃迁的抛弯方向相反, \(v_{i j}\) 的贡献可以用 \(\sigma^{z}\) 描述，

故自旋轨道耦合哈密顿量如下，对其进行傅里叶变换得 \[ \begin{aligned} H_{S O} &=i t_{2} \sum_{i \alpha \beta} \sigma_{\alpha \beta}^{z}\left(S^{z}\right)_{\alpha \beta}\left(c_{i+\vec{c}_{1}, \alpha}^{\dagger} c_{i \beta}-c_{i+\vec{c}_{2}, \alpha}^{\dagger} c_{i \beta}-c_{i-\vec{c}_{1}, \alpha}^{\dagger} c_{i \beta}+c_{i-\vec{c}_{2}, \alpha}^{\dagger} c_{i \beta}\right.\\ &\left.-c_{i+\vec{c}_{1}-\vec{c}_{2}, \alpha}^{\dagger} c_{i \beta}+c_{i-\vec{c}_{1}+\vec{c}_{2}, \alpha}^{\dagger} c_{i \beta}\right) \\ &=i t_{2} \sum_{k \alpha \beta} \sigma_{\alpha \beta}^{z}\left(S^{z}\right)_{\alpha \beta} c_{k \alpha}^{\dagger} c_{k \beta}\left(e^{-i k \cdot \vec{a}_{1}}-e^{-i k \cdot \vec{a}_{2}}-e^{i k \cdot \vec{a}_{2}}+e^{i k \cdot \vec{a}_{2}}\right.\\ &\left.-e^{-i k \cdot\left(\vec{a}_{1}-\vec{a}_{2}\right)}+e^{i k \cdot\left(\vec{a}_{1}-\vec{a}_{2}\right)}\right) \\ &=-4 t_{2} \sum_{k \alpha \beta} \sigma_{\alpha \beta}^{z}\left(S^{z}\right)_{\alpha \beta} c_{k \alpha}^{\dagger} c_{k \beta} \sin \frac{a k_{x}}{2}\left(\cos \frac{a k_{x}}{2}-\cos \frac{\sqrt{3} a k_{y}}{2}\right) \end{aligned} \] 在两个狄拉克点处展开至一阶可以得到 \[ H_{S O}^{\pm}=\pm 3 \sqrt{3} t_{2} \sigma^{z} S^{z} \] 即 \[ H_{S O}=\Delta_{S O} \sigma^{z} \tau^{z} S^{z} \]
